## **#3. 소수점**

```cpp
**//소수점 🤬**
#include <iostream>

using namespace std;

int main() {
    // 변수 선언
    int a = 13;
  	double b = 0.165;
	
		cout << fixed; //소숫점 자리 표현 고정 **🤬🤬**
		cout.precision(6); //6번째 자리까지 반올림 **🤬🤬**
    
    // 출력
    cout << a << " * " << b << " = " << a * b; //다 하나하나 따로따로 << 로 연결해서 적기!
    return 0;
}

**//예시**
#include <iostream>

using namespace std;

int main() {
    double a=5.26, b=8.27;

    cout << fixed;
    cout.precision(3);

    cout << a*b;
    return 0;
}
>>>
43.500
```

## **#4. 변수값 변경**

```cpp
**//같은 변수 값 변경**
#include <iostream>
using namespace std;

	int a = 5;
	cout << "A is " << a << endl;

	a = 3;
	cout << "A is " << a;

	return 0;

}

>>>

A is 3

================================================================================================================

9

================================================================================================================
**//temp; 이용하기 aka. swap!!**
#include <iostream>
using namespace std;

int main() {
    int a = 3, b = 5;
    int temp;

    temp = a; // temp = 3 a=3 b=5
    a = b; // temp=3 a=5 b=5
    b = temp; //temp=3 a=5 b=3

    cout << a << endl;
    cout << b;

    return 0;

}

**//temp 예시1**
#include <iostream>
using namespace std;

int main() {
    int a =2, b=5;

    cout<< a << endl;

    return 0;
}

>>>

2

**//temp 예시2 🤬🤬🤬🤬
```
